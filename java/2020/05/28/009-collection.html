  • кө¬нҳ„мІҙ : HashSet, LinkedHashSet(мҲңм„ң), TreeSet(м •л ¬)

  • лҚ°мқҙн„° мӨ‘ліөм—¬л¶ҖлҘј hashCode()мҷҖ equals()лЎң нҷ•мқё (л§ҒнҒ¬)
    • classлҘј м •мқҳн•  л•Ң лҸҷл“ұк°қмІҙлЎң м Ғмҡ©н•ҳкі  мӢ¶мңјл©ҙ hashCode()мҷҖ equals()мқ„ мҳӨлІ„лқјмқҙл”© н•ҙм•ј н•ңлӢӨ
    • лҸҷл“ұк°қмІҙ : hashCode()мҷҖ equals() лӘЁл‘җ trueмқј л•Ң
  • HashSet : equals(), hashCode() л№„көҗн•ҙм„ң мӨ‘ліөм—¬л¶Җ нҷ•мқё
  • TreeSet : equals(), hashCode() л№„көҗ + Comparable кө¬нҳ„м—¬л¶Җ нҷ•мқё(м •л ¬)

3. Map

  • Map мқён„°нҺҳмқҙмҠӨ
  • мҲңм„ңк°Җ м—ҶлҠ” key-value мҢҚмқҳ лҚ°мқҙн„° м§‘н•©

  • keyк°’мқҖ мӨ‘ліө л¶Ҳк°Җ, лҚ°мқҙн„°лҠ” мӨ‘ліө к°ҖлҠҘ

  • кө¬нҳ„мІҙ : HashMap, LinkedHashMap(мҲңм„ң), TreeMap(м •л ¬), HashTable(лҸҷкё°нҷ” м§Җмӣҗ, кө¬лІ„м „мңјлЎң мӮ¬мҡ© X)
    • HashMap : Entry[]лЎң кҙҖлҰ¬, л°°м—ҙмқҳ indexлҠ” keyк°’мқҳ н•ҙмӢңк°’мңјлЎң кІ°м •
    • LinkedHashMap : л§ҒнҒ¬л“ңлҰ¬мҠӨнҠёлЎң кө¬нҳ„лҗҳм–ҙ мһҲм–ҙ мҲңм„ң ліҙмһҘ
    • TreeMap : R-BнҠёлҰ¬лЎң кҙҖлҰ¬лҗҳм–ҙ н•ӯмғҒ м •л ¬лҗҳм–ҙ мһҲмқҢ, keyк°’мқ„ кё°мӨҖмңјлЎң м •л ¬, NavigableMap кө¬нҳ„
HashMap<Integer, String> map = new HashMap<Integer, String>();
  • HashTable vs HashMap
    • HashMap : лҸҷкё°нҷ”к°Җ м•ҲлҗҳлҠ” нҒҙлһҳмҠӨ(лҸҷмӢңм—җ м—¬лҹ¬ м“°л Ҳл“ңк°Җ м ‘к·ј к°ҖлҠҘ), мӢұкёҖ м“°л Ҳл“ң нҷҳкІҪм—җм„ң лҚ” л№ лҰ„
    • HashTable : лҸҷкё°нҷ”лҘј м§Җмӣҗн•ҳлҠ” нҒҙлһҳмҠӨ(н•ң лІҲм—җ н•ҳлӮҳмқҳ м“°л Ҳл“ңл§Ң м ‘к·ј к°ҖлҠҘ), лӢЁ 5лІ„м „ мқҙмғҒл¶Җн„° м§Җмӣҗн•ҳлҠ” CurrentHashMapмқ„ мӮ¬мҡ©н•ҳлҠ” кІғмқҙ мўӢлӢӨ

  • м»¬л үм…ҳ н”„л Ҳмһ„мӣҢнҒ¬м—җм„ң мӮ¬мҡ©н•ҳлҠ” л°©лІ•
    • Collection мқён„°нҺҳмқҙмҠӨм—җ iterator() м •мқҳ
    • List мҷҖ Set мқён„°нҺҳмқҙмҠӨм—җ iterator() мҳӨлІ„лқјмқҙл”©

ex) Mapм—җм„ң мӮ¬мҡ©н•ҳкё° мң„н•ҙм„ңлҠ” Key, Entry->Set, Value->CollectionмңјлЎң ліҖнҷҳн•ң нӣ„ iterator()мқ„ нҳём¶ңн•ҙм•јн•Ё.
[Map -> Set(Key or Entryмқҳ кІҪмҡ°) : keySet() or entrySet() /В  Map -> Collection(Valueмқҳ кІҪмҡ°) : values() мӮ¬мҡ©н•ҙ ліҖнҷҳ]

public static void main(String[] args) {
		Map<String, Integer> map = new HashMap<String, Integer>();
		
		map.put("лӮҳмһҗл°”", 85); map.put("нҷҚкёёлҸҷ", 90);
		map.put("лҸҷмһҘкө°", 80); map.put("нҷҚкёёлҸҷ", 95);
		
		// keySet()
		Set<String> keySet = map.keySet();
		Iterator<String> keyIterator = keySet.iterator();
		while (keyIterator.hasNext()) {
			String key = keyIterator.next();
			Integer value = map.get(key);
			System.out.println(key + " : " + value);
		}
		System.out.println();
		
		// entrySet()
		Set<Map.Entry<String, Integer>> entrySet = map.entrySet();
		Iterator<Map.Entry<String, Integer>> entryIterator = entrySet.iterator();
		while (entryIterator.hasNext()) {
			Map.Entry<String, Integer> entry = entryIterator.next();
			String key = entry.getKey();
			Integer value = entry.getValue();
			System.out.println(key + " : " + value);
		}
		
		// values()
		Collection<Integer> c = map.values();
		Iterator<Integer> valueIterator = c.iterator();
		while (valueIterator.hasNext()) {
			Integer value = valueIterator.next();
			System.out.println(value);
		}
   }
}
